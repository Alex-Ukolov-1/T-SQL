Create function f4()
returns table
as return
(
select Спортивная_секция.Код_секции,Спортивная_секция.Название_секции, COUNT(*) as sas
From Спортивная_секция
join Запись_в_секции on
Спортивная_секция.Код_секции=Запись_в_секции.Код_секции  
group by  Спортивная_секция.Код_секции,Спортивная_секция.Название_секции
having COUNT(*)>30
)
go
Select * from dbo.f4()
