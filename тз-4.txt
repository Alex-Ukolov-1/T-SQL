Должен быть человек проживающий на Шипиловской улице и одновременно 1990 года рождения
1)
SELECT 
       Читатели.Код_читателя
	   ,Читатели.фамилия
	   ,Читатели.имя
	   ,Читатели.Отчество
	   ,Читатели.дата_рождения
	   ,Читатели.паспортные_данные
	   ,Читатели.адрес
	   ,Читатели.контактный_телефон
FROM dbo.Читатели
WHERE адрес='Шипиловская улица' AND дата_рождения='1990'


2)
SELECT 
	   Читатели.фамилия
	   ,Читатели.контактный_телефон
FROM dbo.Читатели  WITH (nolock)
WHERE фамилия LIKE '%А%' AND фамилия LIKE '%Б%' AND фамилия LIKE '%В%'
